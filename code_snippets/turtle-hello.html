let turtle = turtleConsole.addTurtle()
let scale = 2.5

// H
func h() {
    turtle.rotate(80)
    turtle.forward(scale * 100)
    turtle.arc(radius: scale * 5, angle: 190)
    turtle.forward(scale * 100)
    turtle.rotate(180)
    turtle.forward(scale * 40)
    turtle.arc(radius: scale * 10, angle: -180)
    turtle.forward(scale * 30)
    turtle.arc(radius: scale * 10, angle: 155)
}

// E
func e() {
    turtle.forward(scale * 40)
    turtle.arc(radius: scale * 5, angle: 200)
    turtle.forward(scale * 25)
    turtle.arc(radius: scale * 15, angle: 95)
}

// L
func l() {
    turtle.arc(radius: scale * 15, angle: 80)
    turtle.forward(scale * 85)
    turtle.arc(radius: scale * 5, angle: 190)
    turtle.forward(scale * 82)
    turtle.arc(radius: scale * 15, angle: 90)
}

// O
func o() {
    turtle.arc(radius: scale * 15, angle: 75)
    turtle.forward(scale * 15)
    turtle.arc(radius: scale * 15, angle: -465)
    turtle.arc(radius: scale * 30, angle: 45)
    turtle.penUp()
}


turtle.lineColor(.cyan)
turtle.penDown()

h()
e()
l()
l()
o()

turtle.forward(50)
