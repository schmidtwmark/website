let turtle = turtleConsole.addTurtle()

// Note: flower may be wonky when run at higher speeds
// Radius is not exactly the size of the flower -- size also depends on number of petals
func flower(petals: Int, radius: Double) {
    let petalSpacing = 360.0 / Double(petals)
    for i in 0..<petals {
        turtle.arc(radius: radius, angle: petalSpacing)
        turtle.rotate(180-petalSpacing)
        turtle.arc(radius: radius, angle: petalSpacing)
        turtle.rotate(180)
    }
}

turtle.lineColor(.cyan)
turtle.penDown()
flower(petals: 5, radius: 100)

turtle.penUp()

turtle.forward(200)
turtle.penDown()
flower(petals: 8, radius: 100)
